<script>

// Typing effect
var text="5 March 2025...\nDo you remember this day, Nonaaa?";
var i=0;
function typeWriter(){
if(i<text.length){
document.getElementById("typing").innerHTML+=text.charAt(i);
i++;
setTimeout(typeWriter,60);
}else{
document.getElementById("startBtn").classList.remove("hidden");
}
}
window.onload=typeWriter;

function startStory(){
document.getElementById("story").classList.remove("hidden");
}

function startQuiz(){
document.getElementById("quiz").classList.remove("hidden");
}

function correct(){
document.getElementById("proposal").classList.remove("hidden");
}

function wrong(){
alert("Areeee Nonaaa 😭 Wrong answer! Punishment: Give me one extra hug ❤️");
}

// Stars animation
var canvas=document.getElementById("stars");
var ctx=canvas.getContext("2d");
canvas.width=window.innerWidth;
canvas.height=window.innerHeight;

var stars=[];
for(var i=0;i<100;i++){
stars.push({x:Math.random()*canvas.width,
y:Math.random()*canvas.height,
radius:Math.random()*2});
}

function drawStars(){
ctx.clearRect(0,0,canvas.width,canvas.height);
ctx.fillStyle="white";
for(var i=0;i<stars.length;i++){
ctx.beginPath();
ctx.arc(stars[i].x,stars[i].y,stars[i].radius,0,Math.PI*2);
ctx.fill();
}
}
setInterval(drawStars,50);

// Confetti effect
function confetti(){
for(let i=0;i<100;i++){
let div=document.createElement("div");
div.style.position="fixed";
div.style.width="10px";
div.style.height="10px";
div.style.background="pink";
div.style.top=Math.random()*window.innerHeight+"px";
div.style.left=Math.random()*window.innerWidth+"px";
document.body.appendChild(div);
}
alert("Forever starts now ❤️");
}
